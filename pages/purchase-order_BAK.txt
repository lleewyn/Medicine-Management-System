        window.updateLine = function (idx, sel) {
            const price = parseFloat(sel.options[sel.selectedIndex].getAttribute('data-price')) || 0;
            const row = document.getElementById(`po-line-${idx}`);
            if (!row) return;
            const qtyInput = row.querySelector('input[type=number]');
            const qty = parseFloat(qtyInput.value) || 0;
            const priceEl = document.getElementById(`price-${idx}`);
            const subEl = document.getElementById(`sub-${idx}`);
